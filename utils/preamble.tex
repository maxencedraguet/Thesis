\newcommand\ChapFrame{%
  \AddEverypageHook{%
    \ifodd\value{page}
      \backgroundsetup{contents={%
        \begin{tikzpicture}[overlay,remember picture]
          \node[
            fill=\BoxColor,
            inner sep=2pt,
            rectangle,
            text width=0.8cm,
            text height=3cm,
            align=center,
            anchor=north east
          ] 
          at ($ (current page.north east) + (-0cm,-2*\thechapshift cm) $) 
          {\rotatebox{90}{\hspace*{1.35cm}\parbox[c][1.75cm][t]{3.cm}{%
              \raggedright\textcolor{white}{\\\scshape \thechapter}}}};
        \end{tikzpicture}%
      }%
    }  
    \else
      \backgroundsetup{contents={%
        \begin{tikzpicture}[overlay,remember picture]
        \node[
          fill=\BoxColor,
          inner sep=2pt,
          rectangle,
          text width=0.8cm,
          text height=3cm,
          align=center,
          anchor=north east
        ] 
        at ($ (current page.north east) + (-0cm,-2*\thechapshift cm) $) 
        {\rotatebox{90}{\hspace*{1.35cm}\parbox[c][1.75cm][t]{3.cm}{%
            \raggedright\textcolor{white}{\\\scshape \thechapter}}}};
        \end{tikzpicture}%
      }%
    }  
    \fi
  \BgMaterial}%
  \stepcounter{chapshift}%
}
